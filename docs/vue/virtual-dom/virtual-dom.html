    # Virtual DOM

    # 什么是Virtual DOM

    Virtual DOM(虚拟 DOM)，是由普通的 JS 对象来描述 DOM 对象
    使用 Virtual DOM 来描述真实 DOM

    # 为什么要使用 Virtual DOM

    1. 虚拟 DOM 可以维护程序的状态，跟踪上一次的状态
    2. 通过比较前后两次状态差异更新真实 DOM

    # Snabbdom

    本次虚拟Virtual DOM列子都基于Snabbdom

    1. Vue.js 2.x 内部使用的虚拟 DOM 就是改造的 Snabbdom
    2. 大约 200 SLOC (single line of code)
    3. 通过模块可扩展
    4. 源码使用 TypeScript 开发
    5. 最快的 Virtual DOM 之一

    # 模块的使用

    模块的作用

    1. Snabbdom 的核心库并不能处理 DOM 元素的属性/样式/事件等， 可以通过注册 Snabbdom 默认提供的模块来实现
    2. Snabbdom 中的模块可以用来扩展 Snabbdom的功能
    3. Snabbdom 中的模块的实现是通过注册全局的钩子函数来实现的

    官方提供的模块

    1. attributes 设置Vnode对应的dom属性
    2. props 类似attributes设置dom属性，但是设置属性是通过对象.属性的方式来设置
    3. dataset 处理html5中自定义的data-这种自定义属性
    4. class 切换类样式
    5. style 设置行内样式
    6. eventlisteners 注册事件

    # h 函数

    h 函数的作用就是用来创建VNode对象

    函数重载

    1. 参数个数或参数类型不同的函数
    2. JavaScript 中没有重载的概念
    3. TypeScript 中有重载，不过重载的实现还是通过代码调整参数

    当传入两个参数,则调用第一个add函数,当传入三个参数则调用第三个add函数(js中函数重名会覆盖,所以没有重载的概念)

    当参数都是number类型调用第一个add函数, 当第二个参数是string,调用第二个add函数

  let data: VNodeData = {};
  let children: any;
  let text: any;
  let i: number;
  // 代表一定传的是三个参数
  if (c !== undefined) {
    if (b !== null) {
      data = b;
    }
    // 判断是否是数组 在案例2中 这里传的是一个子元素
    if (is.array(c)) {
      children = c;
      // 判断是否是字符串或者数字 这里显示标签中显示的内容 案例1
    } else if (is.primitive(c)) {
      text = c.toString();
      // c有值并且c有sel这个属性 那么c就是一个vnode对象
    } else if (c && c.sel) {
      children = [c];
    }
    // b的处理逻辑和c处理逻辑一致
  } else if (b !== undefined && b !== null) {
    if (is.array(b)) {
      children = b;
    } else if (is.primitive(b)) {
      text = b.toString();
    } else if (b && b.sel) {
      children = [b];
    } else {
      data = b;
    }
  }
  // 看children是否有值,如果有值,遍历每一个元素
  if (children !== undefined) {
    for (i = 0; i < children.length; ++i) {
      // 判断children的每一项是数字还是字符串,创建文本节点
      if (is.primitive(children[i]))
        children[i] = vnode(
          undefined,
          undefined,
          undefined,
          children[i],
          undefined
        );
    }
  }
  // 判断是否是svg,添加命名空间
  if (
    sel[0] === "s" &&
    sel[1] === "v" &&
    sel[2] === "g" &&
    (sel.length === 3 || sel[3] === "." || sel[3] === "#")
  ) {
    addNS(data, children, sel);
  }
  // 返回Vnode
  return vnode(sel, data, children, text, undefined);
}

    # patch函数

    patch整体过程分析

    1. patch(oldVnode, newVnode)
    2. 把新节点中变化的内容渲染到真实DOM，最后返回新节点作为下一次 处理的旧节点
    3. 对比新旧 VNode 是否相同节点(节点的 key 和 sel 相同)
    4. 如果不是相同节点，删除之前的内容，重新渲染
    5. 如果是相同节点，再判断新的 VNode 是否有 text，如果有并且和 oldVnode的 text不同，直接更新文本内容
    6. 如果新的 VNode有children，判断子节点是否有变化

    接下来我们来分析patch内部的源码

      patch函数就在对应的init函数下 
  /**
   * patch函数接收两个参数，oldVnode 以及 vnode (旧新Vnode)
   * oldVnode既可以是vnode对象也可以是普通的dom对象 在首次渲染时oldVnode需要提供的是一个真实的dom
   */
  return function patch( oldVnode: VNode | Element | DocumentFragment, vnode: VNode): VNode {
    let i: number, elm: Node, parent: Node;
    // 存储新插入节点的队列，把新插入的节点存入到这个队列的目的是为了触发这些节点上设置的insert函数
    const insertedVnodeQueue: VNodeQueue = [];
    // 遍历cps中所有pre中的函数并以此执行 
    for (i = 0; i < cbs.pre.length; ++i) cbs.pre[i]();
    // 判断是否是一个dom对象
    if (isElement(api, oldVnode)) {
      // 如果这是一个dom对象， 通过emptyNodeAt这个函数将dom对象转为vnode对象
      oldVnode = emptyNodeAt(oldVnode);
    }
    // 将dom对象转换为vnode对象后 此时新旧vnode就可以对比了。 
    // sameVnode 判断当前两个节点是否是同一个节点
    if (sameVnode(oldVnode, vnode)) {
      // 如果是相同节点。调用patchVnode 对比这两个内容是否有变化
      // patchVnode函数比较复杂 暂时先不看

    return vnode;

        // 当类样式创建好了之后，会触发create中所有的钩子函数
        for (i = 0; i < cbs.create.length; ++i) cbs.create[i](emptyNode, vnode);

              api.appendChild(elm, createElm(ch as VNode, insertedVnodeQueue));

        } else if (is.primitive(vnode.text)) {

          api.appendChild(elm, api.createTextNode(vnode.text));

      } else {

      }
      // 3. 返回新创建的 DOM 
      return vnode.elm;

    # diff算法

    首先我们需要思考一个问题，在虚拟dom中，为什么要使用diff算法?

    1. 渲染真实dom的开销非常大。dom操作会引起浏览器的重排重绘。 (浏览器的重新渲染)
    2. 浏览器的重新渲染是非常的消耗性能的，因为要重新绘制整个页面。
    3. 虚拟dom中diff的核心就是当数据变化后不会直接去操作dom，而是用js对象描述真实dom，当数据变化后会比较js对象是否发生变化，找到变化后的位置，只取最小化的更新，从而提高性能

    # 执行过程

    # 开始和结束节点

    如果新旧开始节点是 sameVnode (key 和 sel 相同)

    1. 调用 postpatchpatchVnode() 对比和更新节点
    2. 把旧开始和新开始索引往后移动 postpatcholdStartIdx++ / newStartIdx++

    # 旧开始节点 / 新结束节点

    1. 调用 postpatchpatchVnode() 对比和更新节点
    2. 把 postpatcholdStartVnode 对应的 DOM 元素，移动到右边，更新索引

    # 旧结束节点 / 新开始节点

    1. 调用 postpatchpatchVnode() 对比和更新节点
    2. 把 postpatcholdEndVnode 对应的 postpatchDOM 元素，移动到左边，更新索引

    # 非上述情况

    1. 遍历新节点，使用 newStartNode的 key在老节点数组中找相同节点
    2. 如果没有找到，说明 newStartNode是新节点 创建新节点对应的 DOM 元素，插入到 DOM 树中
    3. 如果找到了
    4. 判断新节点和找到的老节点的 postpatchsel 选择器是否相同
    5. 如果不相同，说明节点被修改了，重新创建对应的 DOM 元素，插入到 DOM 树中
    6. 如果相同，把 elmToMove 对应的 DOM 元素，移动到左边

    # 循环结束

    1. 当老节点的子节点个数小于新节点的子节点个数，老节点的所有子节点先遍历完 (oldStartIdx > oldEndIdx)，循环结束

    2. 当老节点的子节点个数大于新节点的子节点个数，新节点的所有子节点先遍历完 (newStartIdx > newEndIdx)，循环结束

    # updateChildren

      /**
   * 
   * @param parentElm 父元素，这是一个dom元素，将来把新创建的dom元素插入到parentElm，或者从parentElm中删除就旧节点对应的dom元素
   * @param oldCh 需要对比的旧节点
   * @param newCh 需要对比的新节点
   * @param insertedVnodeQueue 调用addvnode时需要传入的队列
   */
  function updateChildren(
    parentElm: Node,
    oldCh: VNode[],
    newCh: VNode[],
    insertedVnodeQueue: VNodeQueue
  ) {
    // 旧的开始节点的索引
    let oldStartIdx = 0;
    // 新的开始的索引
    let newStartIdx = 0;
    // 旧的结束节点的索引
    let oldEndIdx = oldCh.length - 1;
    // 旧的开始节点
    let oldStartVnode = oldCh[0];
    // 旧的结束节点
    let oldEndVnode = oldCh[oldEndIdx];
    // 新的结束节点的索引
    let newEndIdx = newCh.length - 1;
    // 新的开始节点
    let newStartVnode = newCh[0];
    // 新的结束节点
    let newEndVnode = newCh[newEndIdx];
    let oldKeyToIdx: KeyToIndexMap | undefined;
    let idxInOld: number;
    let elmToMove: VNode;
    let before: any;

    // 同节点之间的比较
    // 循环的条件: 新旧节点数组中都没有遍历完成
    // 旧节点的开始索引 <= 旧节点的结束索引 && 新节点的开始索引 <= 新节点的结束索引
    while (oldStartIdx <= oldEndIdx && newStartIdx <= newEndIdx) {
      // 首先会先判断新旧节点开始的值是否为null,
      // 当节点是相同节点的时候会对开始和结束节点重新赋值,在赋值的过程中可能存在null的情况
      if (oldStartVnode == null) {
        oldStartVnode = oldCh[++oldStartIdx]; // Vnode might have been moved left
      } else if (oldEndVnode == null) {
        oldEndVnode = oldCh[--oldEndIdx];
      } else if (newStartVnode == null) {
        newStartVnode = newCh[++newStartIdx];
      } else if (newEndVnode == null) {
        newEndVnode = newCh[--newEndIdx];
        // 开始比较开始和结束的4种情况
        // 旧前与新前做对比
      } else if (sameVnode(oldStartVnode, newStartVnode)) {
        // 如果是相同的节点,会对比其中的差异, 进行dom的删除以及添加
        patchVnode(oldStartVnode, newStartVnode, insertedVnodeQueue);
        // 调用完patchVnode后,将新前旧前索引++ 指向下一个节点 (把第二个节点作为开始节点下次循环对比)
        oldStartVnode = oldCh[++oldStartIdx];
        newStartVnode = newCh[++newStartIdx];
        // 如果新前与旧前做对比,不是相同的节点 那么就是判断新后与旧后是否是相同的节点
      } else if (sameVnode(oldEndVnode, newEndVnode)) {
        patchVnode(oldEndVnode, newEndVnode, insertedVnodeQueue);
        // 调用完patchVnode后,将新前旧前索引-- 指向上一个节点 (把倒数第二个节点作为开始节点下次循环对比)
        oldEndVnode = oldCh[--oldEndIdx];
        newEndVnode = newCh[--newEndIdx];
        // 当新后与旧后不是相同的节点 对比新前与旧后是否是相同的节点
      } else if (sameVnode(oldStartVnode, newEndVnode)) {
        // Vnode moved right
        patchVnode(oldStartVnode, newEndVnode, insertedVnodeQueue);
        // 将旧的开始节点的dom元素插入到旧的结束节点的dom元素之后 
        api.insertBefore(
          parentElm,
          oldStartVnode.elm!,
          api.nextSibling(oldEndVnode.elm!)
        );
        // 将旧前索引++ 拿下一个节点  新前索引-- 拿新前上一个节点
        oldStartVnode = oldCh[++oldStartIdx];
        newEndVnode = newCh[--newEndIdx];
        // 如果旧前与新后不是相同的节点,对比旧后与新前
      } else if (sameVnode(oldEndVnode, newStartVnode)) {
        // Vnode moved left
        patchVnode(oldEndVnode, newStartVnode, insertedVnodeQueue);
        // 把旧后对应得dom元素插入(移动)到旧前得的前面
        api.insertBefore(parentElm, oldEndVnode.elm!, oldStartVnode.elm!);
        // 将旧后索引-- 拿上一个节点  新前索引++ 拿新前下一个节点
        oldEndVnode = oldCh[--oldEndIdx];
        newStartVnode = newCh[++newStartIdx];
      } else {
        // 开始和结尾比较结束
        // oldKeyToIdx的作用是存储一个对象，对象的键是老节点的key，值是老节点的索引
        // 根据key找索引
        if (oldKeyToIdx === undefined) {
          oldKeyToIdx = createKeyToOldIdx(oldCh, oldStartIdx, oldEndIdx);
        }
        // 用新节点的key作为属性 找到老节点对应的索引,并赋值给idxInOld这个变量
        idxInOld = oldKeyToIdx[newStartVnode.key as string];
        // 如果是undefined 说明此时新节点在老节点对应的数组中没有元素
        if (isUndef(idxInOld)) {
          // 如果没有的话 通过createElm将新的开始节点转换成dom，然后插入到老开始节点对应的dom元素之前
          api.insertBefore(
            parentElm,
            createElm(newStartVnode, insertedVnodeQueue),
            oldStartVnode.elm!
          );

          // 如果新的开始节点在老的数组节点找到了相同key的老节点
        } else {
          // 将对应的老节点取出来赋值给elmToMove
          elmToMove = oldCh[idxInOld];
          // 将老节点的sel属性和新节点的sel属性比较 是否相同
          // 如果不相同 说明这个节点被修改过
          if (elmToMove.sel !== newStartVnode.sel) {
            // 创建新的开始节点对应的dom元素，把它插入到旧的开始节点对应的dom元素之前
            api.insertBefore(
              parentElm,
              createElm(newStartVnode, insertedVnodeQueue),
              oldStartVnode.elm!
            );

            // 如果新老节点sel相同
          } else {
            // 对比对应的差异 更新真实dom
            patchVnode(elmToMove, newStartVnode, insertedVnodeQueue);
            // 然后把老节点数组中idxInOld处的节点设置为undefined 表示这个节点已经被处理过了
            oldCh[idxInOld] = undefined as any;
            // 然后把elmToMove这个老节点对应dedom元素插入到老节点开始节点的dom元素之前
            api.insertBefore(parentElm, elmToMove.elm!, oldStartVnode.elm!);
          }
        }
        // 新节点处理完毕后将索引++ 获取下一个节点
        newStartVnode = newCh[++newStartIdx];
      }
    }

    // 循环结束的收尾工作
    if (newStartIdx <= newEndIdx) {
      // 老节点数组遍历完 新节点数组如果有剩余
      before = newCh[newEndIdx + 1] == null ? null : newCh[newEndIdx + 1].elm;
      addVnodes(
        parentElm,
        before, // 基准 新增的元素要插入的参考元素
        newCh, // 新节点数组
        newStartIdx, 
        newEndIdx,
        insertedVnodeQueue
      );
    }
    if (oldStartIdx <= oldEndIdx) {
      // 新结束数组遍历完,旧节点有剩余 调用removeVnodes批量删除剩余元素  oldStartIdx, oldEndIdx 剩余节点的位置
      removeVnodes(parentElm, oldCh, oldStartIdx, oldEndIdx);
    }
  }

    2. 在patchVnode函数内部，会先执行用户设置的update钩子函数。 然后判断新节点的text属性。 我们在此处打上断点。
