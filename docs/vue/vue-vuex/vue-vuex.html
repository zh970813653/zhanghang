    # 基本结构

    let _Vue = null

class Store {
  constructor (options) {
    const {
      state = {},
      getters = {},
      mutations = {},
      actions = {}
    } = options
    const store = this
    // vuex中的state是响应式的 所以我们要对vuex中的state进行响应式处理
    store.state = _Vue.observable(state)
    store.getters = Object.create(null)
    Object.keys(getters).forEach(key => {
      Object.defineProperty(store.getters, key, {
        get: () => {
          return getters[key](state)
        }
      })
    })
    store._mutations = mutations
    store._actions = actions
    const { dispatch, commit } = store
    this.dispatch = function boundDispatch (type, payload) {
      return dispatch.call(store, type, payload)
    }
    this.commit = function boundCommit (type, payload) {
      return commit.call(store, type, payload)
    }
  }

  commit (types, payload) {
    console.log(this)
    this._mutations[types](this.state, payload)
  }

  dispatch (types, payload) {
    this._actions[types](this, payload)
  }
}

function install (Vue) {
  _Vue = Vue
  _Vue.mixin({
    beforeCreate () {
      // 首先判断vue实例的options中是否有store， 如果是组件实例 没有store选项，就不需要做这件事情
      if (this.$options.store) {
        _Vue.prototype.$store = this.$options.store
      }
    }
  })
}

export default {
  Store,
  install
}
