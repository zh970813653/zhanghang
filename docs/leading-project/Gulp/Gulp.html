    # Gulp

    # 官方定义

    gulp官方定义为 The steaming build system,也就是基于流的构建系统，之所以选择这种方式，是因为gulp希望实现一个构建管道的概念，这样的话后续做拓展插件的时候就会有统一的方式

    # 安装

    $ npm i Gulp --dev

    # 基本使用

    Gulp往外部导出一个函数可以通过命令来执行这个函数

    $ gulp foo

    当任务执行完毕需要一个回调函数去表示当前任务已经执行完成

    exports.foo = (done) => {
  console.log(1234);
  done() // 表示 任务 完成
}
// 重新执行命令 gulp foo
$ gulp foo

    # 创建组合任务

    通过series、parallel这两个api将多个任务组合成一个组合任务

    const {series,parallel} = require('gulp')
const task1 = done => {
  setTimeout(()=>{
    console.log('task1');
  },1500)
}
const task2 = done => {
  setTimeout(()=>{
    console.log('task2');
  },1500)
}
const task3 = done => {
  setTimeout(()=>{
    console.log('task3');
  },1500)
}
// 执行命令 gulp foo 会依次执行 task1 task2 task3
exports.foo = series(task1,task2,task3)
// 执行命令 gulp foo 会同时执行 task1 task2 task3
exports.foo = parallel(task1,task2,task3)

    总结

    1. 总的来说，创建并行任务或者串行任务在实际构建工作流时是非常有用的
    2. 比如，编译css时与编译js时，它们之间是互不干扰的，这种情况可以通过并行任务方式去执行，这样会提高构建效率
    3. 再比如，我们部署的时候需要先执行编译任务， 这种情况下可以通过series串行的模式去执行任务

    # 构建过程核心原理

    代码示例,将css代码进行压缩

    const fs = require('fs')
const {Transform} = require('stream')
exports.default = () => {
  // 文件读取流
  const read = fs.createReadStream('index.css')
  // 文件写入流
  const write = fs.createWriteStream('index.min.css')
  // 文件转换流
  const transform = new Transform({
    transform: (chunk,encoding,callback)=>{
      // 核心转换过程
      // chunk => 读取流中读取到的内容 (bugger)
      const input = chunk.toString()
      console.log(input);
      // 替换所有空格以及注释
      const output = input.replace(/\s+/g,'').replace(/\/\*.+?\//g,'')
      callback(null,output)
    }
  })
  read
  .pipe(transform) // 转换
  .pipe(write) // 写入
  return read
}

    # 自动加载插件

    自动加载插件

    从以上代码看出，随着项目越来越大越来越复杂，如果都是通过手动方式去载入插件的话，require操作会非常的多，非常不利于后期维护代码。
    可以通过一个插件去解决这个问题gulp-load-plugins

    // 安装gulp-load-plugins，并对所有的插件后缀名称进行修改
$ npm i gulp-load-plugins --sabv-dev

const { src, dest, parallel,series } = require('gulp')
const loadPlugins = require('gulp-load-plugins')
const plugins = loadPlugins()
const del = require('del')
const sass = require('gulp-sass')(require('sass'))
const options = {
  base: 'src'
}
const style = () => {
  return src('src/assets/styles/*.scss', options)
    .pipe(sass())
    .pipe(dest('dist'))
}
const clean = ()=> {
  return del('dist')
}
const script = () => {
  return src('src/assets/scripts/*.js', options)
    .pipe(plugins.babel({ presets: ['@babel/preset-env'] }))
    .pipe(dest('dist'))
}
const pages = () => {
  // 同时也可以src/**/*.html  **表示所有的目录下
  return src('src/*.html', options)
    .pipe(plugins.swig())
    .pipe(dest('dist'))
}
const image = () => {
  return src('src/assets/images/**', options)
    .pipe(plugins.imagemin())
    .pipe(dest('dist'))
}
const public = () => {
  return src('public/**', {base:'public'})
  .pipe(dest('dist'))
}
const font = () => {
  return src('src/assets/fonts/**', options)
    .pipe(plugins.imagemin())
    .pipe(dest('dist'))
}
const compile = parallel(style, script, pages,image,font)
const build = parallel(compile,public)
const cleanBuild = series(clean,build)
module.exports = {
  compile,
  build,
  cleanBuild
}

    gulp-load-plugins的坑

    1. sass没法替换 因为最新的sass不支持node-sass，gulp-load-plugins未做到兼容
    2. 在通过plugins.xxx()使用某个插件时 会从package.json文件中的依赖找这个包，本地安装但是没有存到package.json文件中的devDependencies或者dependencies里面，它是不会从node_modules中加载的。所以在安装插件时一定要指定下载环境。

    # 热更新开发服务器

    我们需要一个本地的开发服务器，来辅助我们，提高开发效率,可以使用browser-sync

// 引入browser-sync 并通过create创建
const browserSync = require('browser-sync')
const bs = browserSync.create()
const dev = ()=> {
  bs.init({
    notify: false, // 删除网页打开右上角提示
    port: 2080, // 设置服务器打开端口
    open: true, // 默认true 自动打开浏览器

      routes: {
        '/node_modules': 'node_modules' // 引入的文件前置是node_modules的情况下会默认引入node_modules下对应地址的目录
      }

    }
  })
}
module.exports = {
  dev

    # watch

    以上方法，只能通过修改dist让浏览器重新刷新，如果我们需要修改src目录下的文件，从而触发浏览器的更新，我们要怎么做呢?

    // 利用gulp中自带的api watch
const { src, dest, parallel,series,watch } = require('gulp')
const loadPlugins = require('gulp-load-plugins')
const plugins = loadPlugins()
const del = require('del')
const sass = require('gulp-sass')(require('sass'))
const browserSync = require('browser-sync')
const bs = browserSync.create()
const options = {
  base: 'src'
}
const style = () => {
  return src('src/assets/styles/*.scss', options)
    .pipe(sass())
    .pipe(dest('dist'))
    .pipe(bs.reload({stream:true})) // stream:true 将文件以流的方式传输
}
const clean = ()=> {
  return del('dist')
}
const script = () => {
  return src('src/assets/scripts/*.js', options)
    .pipe(plugins.babel({ presets: ['@babel/preset-env'] }))
    .pipe(dest('dist'))
    .pipe(bs.reload({stream:true}))
}
const pages = () => {
  // 同时也可以src/**/*.html  **表示所有的目录下
  return src('src/*.html', options)
    .pipe(plugins.swig())
    .pipe(dest('dist'))
    .pipe(bs.reload({stream:true}))
}
const image = () => {
  return src('src/assets/images/**', options)
    .pipe(plugins.imagemin())
    .pipe(dest('dist'))
}
const public = () => {
  return src('public/**', {base:'public'})
  .pipe(dest('dist'))
}
const font = () => {
  return src('src/assets/fonts/**', options)
    .pipe(plugins.imagemin())
    .pipe(dest('dist'))
}
const dev = ()=> {
  watch(('src/assets/images/**'),style)
  watch(('src/assets/scripts/*.js'),script)
  watch(('src/*.html'),pages)
  watch([
    'src/assets/images/**',
    'public/**',
    'src/assets/fonts/**'
  ],bs.reload)  // bs.reload 重新更新浏览器 浏览器重新发起请求 拿到最新的文件 

  bs.init({
    notify: false, // 删除网页打开右上角提示
    port: 2080, // 设置服务器打开端口
    open: true, // 默认true 自动打开浏览器
    // files: 'dist/**', 
    server: {
      baseDir: ['dist','src','public'], 
      routes: {
        '/node_modules': 'node_modules'
      }

    }
  })
}
const compile = parallel(style, script, pages)
const build = parallel(compile,image,font,public)
// 生成打包执行的任务
const cleanBuild = series(clean,build)
// 开发环境下执行的任务
const develop = series(compile,dev)
module.exports = {
  compile,
  build,
  cleanBuild,
  develop
}

    代码解析：我们在dev指令函数下watch去指定监听一个路径对应的文件，当文件进行变化后，会主动执行后面指定的任务函数。 在这过程当中我们需要思考哪些任务只需要在开发阶段执行，哪些任务不需要在开发阶段执行，列如上述代码的image它的作用只是做了一个画质无损的压缩，所以我们可以在开发阶段减少一次构建。
    值得注意的是： 我们在指定的style、script、pages下添加了新的pipe方法 pipe(bs.reload({stream:true})。它的作用是重新发起对文件新数据的请求。
    同时，server配置项中的baseDir指定为一个数组。 让请求的资源先从初始的位置依次往后查找
    完整流程：watch监听路径文件变化，当文件发生改变后触发watch第二个参数指定的方法函数。 从而触发bs.reload重新请求新资源，从而触发浏览器更新

    # 重新规划构建过程

    通过以上压缩的案列。 我们发现一个问题。在实际开发项目工作中。我们需要的是将dist目录上线。 可是压缩代码的过程中，我们只是将dist里面html文件压缩后重新放到了release目录下。而dist目录本身html并没有得到压缩。 所以我们需要重新规划构建过程，对压缩的代码进行修改。

    // 修改过后的完整代码
const { src, dest, parallel,series,watch } = require('gulp')
const loadPlugins = require('gulp-load-plugins')
const plugins = loadPlugins()
const del = require('del')
const sass = require('gulp-sass')(require('sass'))
const browserSync = require('browser-sync')
const bs = browserSync.create()
const options = {
  base: 'src'
}
const style = () => {
  return src('src/assets/styles/*.scss', options)
    .pipe(sass())
    .pipe(dest('temp'))
    .pipe(bs.reload({stream:true})) // stream:true 将文件以流的方式传输
}
const clean = ()=> {
  return del(['dist,temp'])
}
const script = () => {
  return src('src/assets/scripts/*.js', options)
    .pipe(plugins.babel({ presets: ['@babel/preset-env'] }))
    .pipe(dest('temp'))
    .pipe(bs.reload({stream:true}))
}
const pages = () => {
  // 同时也可以src/**/*.html  **表示所有的目录下
  return src('src/*.html', options)
    .pipe(plugins.swig())
    .pipe(dest('temp'))
    .pipe(bs.reload({stream:true}))
}
const image = () => {
  return src('src/assets/images/**', options)
    .pipe(plugins.imagemin())
    .pipe(dest('dist'))
}
const public = () => {
  return src('public/**', {base:'public'})
  .pipe(dest('dist'))
}
const font = () => {
  return src('src/assets/fonts/**', options)
    .pipe(plugins.imagemin())
    .pipe(dest('dist'))
}
const dev = ()=> {
  watch(('src/assets/images/**'),style)
  watch(('src/assets/scripts/*.js'),script)
  watch(('src/*.html'),pages)
  watch([
    'src/assets/images/**',
    'public/**',
    'src/assets/fonts/**'
  ],bs.reload)  // bs.reload 重新更新浏览器 浏览器重新发起请求 拿到最新的文件 

  bs.init({
    notify: false, // 删除网页打开右上角提示
    port: 2080, // 设置服务器打开端口
    open: true, // 默认true 自动打开浏览器
    server: {
      baseDir: ['temp','src','public'], 
    }
  })
}
const useref = () => {
  return src('temp/*.html',{base:'temp'})
  .pipe(plugins.useref({searchPath:['temp','.']}))
  .pipe(plugins.if(/.js$/,plugins.uglify())) // 判断读取流是不是以js为结尾，然后调用对应的插件
  .pipe(plugins.if(/.css$/,plugins.cleanCss()))
  .pipe(plugins.if(/.html$/,plugins.htmlmin({
    collapseWhitespace:true, // 是否删除空格
    minifyCSS: true, // 是否将html中style标签中css的压缩
    minifyJS: true, // 是否将html中style标签中js的压缩
  })))
  .pipe(dest('dist'))
}
const compile = parallel(style, script, pages)
const build = parallel(series(compile,useref),image,font,public)
// 生成打包执行的任务
const cleanBuild = series(clean,build)
// 开发环境下执行的任务
const develop = series(compile,dev)
module.exports = {
  compile,
  build,
  cleanBuild,
  develop,
  useref
}
